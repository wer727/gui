// 洗牌函数，返回一个随机排序的扑克牌数组 function shuffleDeck(deck) { for (let i = 0; i < deck.length; i++) { let j = Math.floor(Math.random() * (i + 1)); let temp = deck[i]; deck[i] = deck[j]; deck[j] = temp; } return deck; } // 发牌函数，返回两个玩家手中的牌 function dealCards(deck) { let player1Hand = []; let player2Hand = []; for (let i = 0; i < deck.length; i++) { if (i % 2 === 0) { player1Hand.push(deck[i]); } else { player2Hand.push(deck[i]); } } return [player1Hand, player2Hand]; } // 比大小函数，返回胜者的名字 function compareCards(player1Card, player2Card) { const values = [ "Ace", "2", "3", "4", "5", "6", "7", "8", "9", "10", "Jack", "Queen", "King", ]; const player1Value = values.indexOf(player1Card.split(" ")[0]) + 1; const player2Value = values.indexOf(player2Card.split(" ")[0]) + 1; if (player1Value > player2Value) { return "Player 1"; } else if (player2Value > player1Value) { return "Player 2"; } else { return "Tie"; } } // 测试 let shuffledDeck = shuffleDeck(deck); let playerHands = dealCards(shuffledDeck); let player1Card = playerHands[0][0]; let player2Card = playerHands[1][0]; console.log("Player 1's card: " + player1Card);
